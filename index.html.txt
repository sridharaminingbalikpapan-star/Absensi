<script>
const api = 'URL_WEB_APP_SCRIPT';
new Html5Qrcode('reader').start(
{ facingMode: 'environment' },
{ fps: 10, qrbox: 250 },
qr => {
fetch(api, {
method:'POST',
body:JSON.stringify({id:qr})
})
.then(r=>r.json())
.then(res=>{
document.getElementById('result').innerHTML = `${res.nama} - ${res.status}`;
});
}
);
</script>
